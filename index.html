[7 kyu] Which color is the brightest?

                
let brightest = colors => { 
    let rgbSeparatedArr = [], maxArr = []; 
    for (let i = 0; i < colors.length; i++) { 
        rgbSeparatedArr.push(colors[i].match(/[0-9A-F]{2}/g)); 
        maxArr.push(Math.max(...(rgbSeparatedArr[i].map(x => parseInt(x, 16))))); 
    }
    return colors[maxArr.indexOf(Math.max(...maxArr))]; 
}

[7 kyu] Digits explosion

                
function explode(s) { 
    return s.split("").map(x => x.repeat(x)).join(""); 
}

[7 kyu] Isograms

                
let isIsogram = str => { 
    let set = new Set(str.toLowerCase().split("")); 
    return set.size == str.length ? true : false; 
}
